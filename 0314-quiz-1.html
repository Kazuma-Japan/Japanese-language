  <script>

    // 問題データ配列（CSVファイルの内容そのまま）

    const quizData = [

      {

        question: "apple",

        meaning: "りんご",

        examples: [

          "I eat an apple.",

          "The apple is red."

        ]

      },

      {

        question: "book",

        meaning: "本",

        examples: [

          "This is a book.",

          "I read a book."

        ]

      },

      {

        question: "cat",

        meaning: "猫",

        examples: [

          "The cat is sleeping.",

          "The cat likes fish."

        ]

      },

      {

        question: "dog",

        meaning: "犬",

        examples: [

          "The dog is barking.",

          "I walk the dog."

        ]

      },

      {

        question: "elephant",

        meaning: "象",

        examples: [

          "The elephant is big.",

          "I saw an elephant."

        ]

      }

    ];

    

    // DOM要素取得

    const questionText = document.getElementById('questionText');

    const meaningText = document.getElementById('meaningText');

    const examplesList = document.getElementById('examplesList');

    const counter = document.getElementById('counter');

    const progressBar = document.getElementById('progressBar');

    const answerArea = document.getElementById('answerArea');

    const showAnswerBtn = document.getElementById('showAnswerBtn');

    const nextBtn = document.getElementById('nextBtn');

    const questionArea = document.getElementById('questionArea');

    const completedArea = document.getElementById('completedArea');

    const restartBtn = document.getElementById('restartBtn');

    

    // クイズ管理変数

    let currentQuestions = [];

    let currentIndex = 0;

    

    // クイズを初期化する

    function initQuiz() {

      // ランダムに問題を選ぶ

      currentQuestions = [...quizData].sort(() => Math.random() - 0.5);

      currentIndex = 0;

      

      // 初期表示

      updateProgressBar();

      updateCounter();

      showQuestion();

      

      // 表示エリアの切り替え

      questionArea.style.display = 'block';

      completedArea.style.display = 'none';

      completedArea.classList.remove('active');

    }

    

    // 質問を表示する

    function showQuestion() {

      const currentQuestion = currentQuestions[currentIndex];

      

      // 質問テキスト設定

      questionText.textContent = currentQuestion.question;

      questionText.classList.add('pulse');

      

      // アニメーション終了後にクラスを削除

      setTimeout(() => {

        questionText.classList.remove('pulse');

      }, 1000);

      

      // 答え非表示

      answerArea.classList.remove('active');

      

      // ボタン状態設定

      showAnswerBtn.style.display = 'flex';

      nextBtn.style.display = 'none';

    }

    

    // 答えを表示する

    function showAnswer() {

      const currentQuestion = currentQuestions[currentIndex];

      

      // 意味テキスト設定

      meaningText.textContent = `【意味】 ${currentQuestion.meaning}`;

      

      // 例文リスト作成

      examplesList.innerHTML = '';

      currentQuestion.examples.forEach(example => {

        const li = document.createElement('li');

        li.textContent = example;

        examplesList.appendChild(li);

      });

      

      // 答え表示

      answerArea.classList.add('active');

      

      // ボタン状態設定

      showAnswerBtn.style.display = 'none';

      nextBtn.style.display = 'flex';

    }

    

    // 次の問題へ進む

    function nextQuestion() {

      currentIndex++;

      

      if (currentIndex < currentQuestions.length) {

        updateProgressBar();

        updateCounter();

        showQuestion();

      } else {

        // 全問終了

        completeQuiz();

      }

    }

    

    // クイズ終了処理

    function completeQuiz() {

      questionArea.style.display = 'none';

      completedArea.style.display = 'block';

      

      // アニメーション表示

      setTimeout(() => {

        completedArea.classList.add('active');

      }, 100);

    }

    

    // プログレスバーを更新する

    function updateProgressBar() {

      const progress = (currentIndex / currentQuestions.length) * 100;

      progressBar.style.width = `${progress}%`;

    }

    

    // カウンターを更新する

    function updateCounter() {

      counter.textContent = `${currentIndex + 1}/${currentQuestions.length}`;

    }

    

    // イベントリスナー設定

    showAnswerBtn.addEventListener('click', showAnswer);

    nextBtn.addEventListener('click', nextQuestion);

    restartBtn.addEventListener('click', initQuiz);

    

    // クイズ開始

    initQuiz();

  </script>
